#Pulses
on_birthday = {
	on_actions = {
		fire_exp_on_action
		water_exp_on_action
		air_exp_on_action
		earth_exp_on_action
		a_spiritualiity_exp_on_action
	}
}

#Bending Experience

fire_exp_on_action = {
	trigger = {
		has_trait = lifestyle_fire_bender
		NOT = {
		    has_trait_xp = {
		    	trait = lifestyle_fire_bender
			value >= 100
			}
		}
	}
	effect = {
		random_list = {
			10 = {
				add_trait_xp = {
					trait = lifestyle_fire_bender
					value = 1
				}
			}
			2 = {
				add_trait_xp = {
					trait = lifestyle_fire_bender
					value = 2
				}
			}
		}
	}
}
