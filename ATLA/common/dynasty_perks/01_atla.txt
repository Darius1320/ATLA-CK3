# BENDING LEGACY

bending_legacy_1 = { # Born Commanders
	legacy = bending_legacy_track

	can_be_picked = {
		OR = {
                has_trait_xp = {
                    trait = lifestyle_fire_bender
                    value >= 30
                }
                has_trait_xp = {
                    trait = lifestyle_water_bender
                    value >= 30
                }
                has_trait_xp = {
                    trait = lifestyle_air_bender
                    value >= 30
                }
                has_trait_xp = {
                    trait = lifestyle_earth_bender
                    value >= 30
				}
			}
		traits = { bending_commander = 100 }
	}

		effect = {
		# Tell the player they'll unlock the elemental commander trait
		custom_description_no_bullet = { text = bending_legacy_1_ct_effect }
	}

	character_modifier = {
		county_opinion_add = 5
		monthly_county_control_growth_add = 0.1
	}

	ai_chance = {
		value = 11
		if = {
			limit = {
				dynasty = {
					dynasty_prestige_level >= high_dynasty_prestige_level
				}
			}
			multiply = 10
		}
		if = {
			limit = {
				can_start_new_legacy_track_trigger = no
			}
			multiply = 0
		}
	}
}
